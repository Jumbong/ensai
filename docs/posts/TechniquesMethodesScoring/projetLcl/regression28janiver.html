def calculate_woe_iv(dataset, feature, target):
    lst = []
   
    for i in range(dataset[feature].nunique()):
        val = list(dataset[feature].unique())[i]
        lst.append({
            'Bin Values': val,
            'All': dataset[dataset[feature] == val].count()[feature],
            'Good': dataset[(dataset[feature] == val) & 
                    (dataset[target] == 0)].count()[feature],
            'Bad': dataset[(dataset[feature] == val) & 
                    (dataset[target] == 1)].count()[feature]
        }) 
    dset = pd.DataFrame(lst)
    dset['Distr_Good'] = dset['Good'] / dset['Good'].sum()
    dset['Distr_Bad'] = dset['Bad'] / dset['Bad'].sum()
    dset['WoE'] = np.log(dset['Distr_Good'] / dset['Distr_Bad'])
    dset = dset.replace({'WoE': {np.inf: 0, -np.inf: 0}})
    dset['IV'] = (dset['Distr_Good'] - dset['Distr_Bad']) * dset['WoE']
    iv = dset['IV'].sum()
    dset = dset.sort_values(by='WoE')
    return dset, iv

import statsmodels.formula.api as smf

def feature_selection_logistic_backward_formula(data, target, criterion='aic'):
    all_features = data.columns.tolist()
    all_features.remove(target)
    best_criteria_value = np.inf
    criterion = 'aic'  # ou 'bic' selon le critère que vous voulez utiliser
    
    while all_features:
        worst_feature = None
        current_criteria_value = np.inf  # Initialiser avec une valeur infinie à chaque itération
    
        for feature in all_features:
            # Créer la formule sans la caractéristique courante
            remaining_features = [f for f in all_features if f != feature]
            formula = f"{target} ~ " + " + ".join(remaining_features)
            
            # Construire et ajuster le modèle
            model = smf.logit(formula, data=data).fit(disp=0)
            
            # Utiliser AIC ou BIC comme critère
            criteria_value = model.bic if criterion == 'bic' else model.aic
    
            # Vérifier si le retrait de cette caractéristique améliore le modèle
            if criteria_value < current_criteria_value:
                current_criteria_value = criteria_value
                worst_feature = feature
    
        if worst_feature and current_criteria_value < best_criteria_value:
            all_features.remove(worst_feature)
            best_criteria_value = current_criteria_value
            print(f"Removed feature: {worst_feature}, {criterion.upper()}: {best_criteria_value:.4f}")
        else:
            # Arrêtez la boucle si aucune amélioration n'est trouvée
            break
    return all_features

import statsmodels.formula.api as smf
import pandas as pd

def backward_selection_logistic(data, target, criterion='aic'):
    """
    Perform backward selection of predictors using logistic regression.
    
    :param data: pandas DataFrame with all variables
    :param target: string, name of the target variable in the DataFrame
    :param criterion: 'aic' or 'bic', criterion to use for model comparison
    :return: list of selected predictors
    """
    remaining_predictors = set(data.columns)
    remaining_predictors.remove(target)
    
    best_criterion_value = float('inf')
    best_model = None

    while remaining_predictors:
        models_with_predictors = []
        for predictor in remaining_predictors:
            formula = f"{target} ~ " + " + ".join(remaining_predictors - {predictor})
            model = smf.logit(formula, data=data).fit(disp=False)
            criterion_value = model.aic if criterion == 'aic' else model.bic
            models_with_predictors.append((criterion_value, predictor, model))

        models_with_predictors.sort()
        #print(models_with_predictors)
        best_new_criterion_value, best_predictor_to_remove, best_new_model = models_with_predictors[0]
        
        if best_new_criterion_value < best_criterion_value:
            print(f"Removing {best_predictor_to_remove}, Criterion {criterion}: {best_new_criterion_value}")
            remaining_predictors.remove(best_predictor_to_remove)
            best_criterion_value = best_new_criterion_value
            best_model = best_new_model
        else:
            break

    print("Selected predictors:", remaining_predictors)
    return list(remaining_predictors)
